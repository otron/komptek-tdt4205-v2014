	\node							(a)		{a};
	\node[right of=a, fill=green]	(b)		{b};
	\node[right of=b]				(c)		{c};
	\node[below of=a, fill=green]	(d)		{d};
	\node[right of=d, fill=red]		(e)		{e};

	\path
	(a) 	edge 	(b)
			edge	(d)
			edge	(e)
	(b)		edge 	(c)
	(d)		edge	(e);
